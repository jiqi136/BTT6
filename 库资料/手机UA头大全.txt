import re  # 正则式

def 读取文件():
    try:  # 调用异常处理,应对易发生错误的位置
        文本 = open(r"E:\PY学习文件\BTT影视剧\库资料\手机UA头大全.txt", 'r', encoding='UTF-8')
        文本内容列表 = 文本.readlines()  # read() #全部读取   readlines 每一行
    except UnicodeDecodeError as 异常原因:  # 异常处理
        print(异常原因)
        文本 = open(r"E:\PY学习文件\BTT影视剧\py快捷方式\代理IP.txt", 'r', encoding='gbk')
        文本内容列表 = 文本.readlines()  # read() #全部读取   readlines每一行
    else:  # 必须放在所有的except子句之后.这个子句将在try子句没有发生任何异常的时候执行.
        文本.close()

    return 文本内容列表#返回


文本内容列表=读取文件()
随机浏览器列表=[]

for 文本内容 in 文本内容列表:


    规则='ndroid.{2,}? '
    文本内容 =re.sub(规则, 'ndroid {安卓系统版本号} ', 文本内容)#替换   ,count=0,re.S|re.I

    规则 = 'hone .{0,}? like'
    文本内容 = re.sub(规则, 'hone {IP版本号} like', 文本内容)  # 替换   ,count=0,re.S|re.I

    规则 = 'Messenger/.{0,}? '
    文本内容 = re.sub(规则, 'Messenger/{微信版本号} ', 文本内容)  # 替换   ,count=0,re.S|re.I



    规则 = '/.{0,5}Mobile'
    文本内容 = re.sub(规则, '/{浏览器版本号} Mobile', 文本内容)  # 替换   ,count=0,re.S|re.I

    规则 = '/.{0,5}TBS'
    文本内容 = re.sub(规则, '/{浏览器版本号} TBS', 文本内容)  # 替换   ,count=0,re.S|re.I

    规则 = '/.{0,5}QQ-'
    文本内容 = re.sub(规则, '/{浏览器版本号} QQ-', 文本内容)  # 替换   ,count=0,re.S|re.I

    规则 = 'Browser/.{0,5}/Adr'
    文本内容 = re.sub(规则, 'Browser/{浏览器版本号}/Adr', 文本内容)  # 替换   ,count=0,re.S|re.I

    规则 = 'Browser/.{0,5} '
    文本内容 = re.sub(规则, 'Browser/{浏览器版本号} ', 文本内容)  # 替换   ,count=0,re.S|re.I


    规则 = 'Chrome/.{0,}? '
    文本内容 = re.sub(规则, 'Chrome/{Chrome浏览器版本号} ', 文本内容)  # 替换   ,count=0,re.S|re.I


    if '版本号' in 文本内容 and 'Mac' not in 文本内容 and 'Windows' not in 文本内容:
        文本内容 = 文本内容.replace("\n", "") #替换   , 1) 次数 1
        随机浏览器列表.append(文本内容)  # 列表追加
